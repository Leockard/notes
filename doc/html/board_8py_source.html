78  def GetNextCard(self, card, direc):
79  """
80  Returns the nearest card to card in the direction direc, which must be one of
81  Board.LEFT, Board.RIGHT, Board.UP, or Board.DOWN.
82  """
83  # depending on the direction we compare a different side
84  # of the cards, as well as get the points whose distance
85  # we're going to calculate in a different way
86  if direc == "left":
87  side = lambda x: x.right
88  getp1 = lambda x: x.GetTopLeft()
89  getp2 = lambda x: x.GetBottomLeft()
90  elif direc == "right":
91  side = lambda x: x.left
92  getp1 = lambda x: x.GetTopLeft()
93  getp2 = lambda x: x.GetTopRight()
94  elif direc == "up":
95  side = lambda x: x.bottom
96  getp1 = lambda x: x.GetTopLeft()
97  getp2 = lambda x: x.GetBottomLeft()
98  elif direc == "down":
99  side = lambda x: x.top
100  getp1 = lambda x: x.GetBottomLeft()
101  getp2 = lambda x: x.GetTopLeft()
102 
103  # get those cards whose "side" is in the desired position with respect to card
104  rect = card.GetRect()
105  nxt = []
106  if direc == "left" or direc == "up":
107  nxt = [c for c in self.GetCards() if side(c.GetRect()) < side(rect)]
108  elif direc == "right" or direc == "down":
109  nxt = [c for c in self.GetCards() if side(c.GetRect()) > side(rect)]
110  else:
111  return None
112 
113  # we're going to use getp1 to get a point in card and compare
114  # it to the point got by getp2 on all the cards in nxt
115  if nxt:
116  # order them by distance
117  nxt.sort(key=lambda x: dist2(getp1(x.GetRect()), getp2(rect)))
118  # and return the nearest one
119  return nxt[0]
120  else:
121  return None

149  """
150  Places a new Card on the board.
151  subclass should be the string with the name of the Card subclass to create.
152  below=False creates the new Card to the right of the currently selected

166  if below:
167  top = rect.bottom + pad
168  left = rect.left
169  else:
170  top = rect.top

174  # if cursor is inside a card, place it next to it
175  elif GetCardAncestor(self.FindFocus()):
176  rect = GetCardAncestor(self.FindFocus()).GetRect()

188  top = min([r.top for r in rects])

196  def NewCard(self, subclass, pos=wx.DefaultPosition, scroll=False):
197  """
198  Create a new card of type subclass (string) at pos. If scroll
199  is True, scroll the board so that the new card is in view.
200  """
201  # never use labels, always let Board set its own
202  label = len(self.cards)
203 
204  # create the new card with the unscaled position
205  # so that we can just call new.Stretch() afterward
206  # to set both position and size
207  pos = [i / self.scale for i in pos]
208 
209  if subclass == "Content":
210  new = Content(self, label, pos=pos)
211  elif subclass == "Header":
212  new = Header(self, label, pos=pos)
213  elif subclass == "Image":
214  new = Image(self, label, pos=pos)
215  new.Stretch(self.scale)
216 
217  # set bindings for every card
218  new.Bind(wx.EVT_LEFT_DOWN, self.OnCardLeftDown)
219  new.Bind(wx.EVT_CHILD_FOCUS, self.OnCardChildFocus)
220  new.Bind(Card.EVT_CARD_DELETE, self.OnCardDelete)
221  new.Bind(Card.EVT_CARD_COLLAPSE, self.OnCardCollapse)
222  new.Bind(Card.EVT_CARD_REQUEST_INSPECT, self.OnCardRequest)
223  for ch in new.GetChildren():
224  ch.Bind(wx.EVT_LEFT_DOWN, self.OnCardChildLeftDown)
225 
226  # raise the appropriate event
227  event = self.NewCardEvent(id=wx.ID_ANY, subclass=subclass)
228  event.SetEventObject(new)
229  self.GetEventHandler().ProcessEvent(event)
230 
231  # make enough space and breathing room for the new card
232  self.FitToChildren()
233  self.ExpandVirtualSize(self.GetPadding() * 2, self.GetPadding() * 2)
234 
235  # make sure the new card is visible
236  if scroll:
237  rect = new.GetRect()
238  board = self.GetRect()
239  if rect.bottom > board.bottom or rect.right > board.right or rect.left < 0 or rect.top < 0:
240  self.ScrollToCard(new)
241 
242  # finish up
243  new.SetFocus()
244  self.cards.append(new)
245  return new

348  def ScrollToPoint(self, pt):
349  """
350  If the point is in view, don't do anything. Otherwise, scroll it into view.
351  The point must be in absolute (content size) coordinates.
352  """
353  step = self.SCROLL_STEP
354 
355  # get the current rect in view, in pixels
356  # coordinates relative to underlying content size
357  view = [k * step for k in self.GetViewStart()]
358  sz = self.GetClientSize()
359  rect = wx.Rect(view[0], view[1], sz.width, sz.height)
360 
361  # point we're scrolling to (already in pixels)
362  # relative to content size
363 
364  # nothing to do
365  if rect.Contains(pt):
366  return
367 
368  # scroll the point into view
369  scroll = False
370  pad = self.GetPadding()
371 
372  # if one of the argumets is wx.DefaultCoord,
373  # we will not scroll in that direction
374  ysc = wx.DefaultCoord
375  xsc = wx.DefaultCoord
376 
377  # remember y coordinate grows downward
378  if pt.x >= rect.right or pt.x <= rect.left:
379  scroll = True
380  xsc = pt.x - pad # where we want to go
381  xsc /= step # in scroll units
382  if pt.y <= rect.top or pt.y >= rect.bottom:
383  scroll = True
384  ysc = pt.y - pad # where we want to go
385  ysc /= step # in scroll units
386 
387  if scroll:
388  # will scroll as much as it's possible
389  # i.e., pt will not necessarily be in the top left corner after scrolling
390  # but it will surely be inside the view
391  self.Scroll(xsc, ysc)

400  def HArrangeSelectedCards(self):
401  """
402  If there are any selected cards, arrange them in a horizontal row,
403  to the right of the left-most selected card.
404  """
405  if len(self.GetSelection()) < 1: return
406 
407  # we unselect first so that we erase the selection rectangles correctly
408  arrange = self.GetSelection()[:]
409  self.UnselectAll()
410 
411  lefts = [c.GetRect().left for c in arrange]
412  left = min(lefts)
413  card = arrange[lefts.index(left)]
414  top = card.GetRect().top
415  arrange.sort(key=lambda x: x.GetRect().left)
416 
417  for c in arrange:
418  c.SetPosition(wx.Point(left, top))
419  left = c.GetRect().right + self.GetPadding()
420 
421  self.FitToChildren()
422  self.selec.SetFocus()

424  def VArrangeSelectedCards(self):
425  """
426  If there are any selected cards, arrange them in a vertical column,
427  below of the top-most selected card.
428  """
429  if len(self.GetSelection()) < 1: return
430 
431  # value-copy the list since we may do weird things to it
432  arrange = self.GetSelection()[:]
433 
434  # compute the pivot
435  tops = [c.GetRect().top for c in arrange]
436  top = min(tops)
437  card = arrange[tops.index(top)]
438  left = card.GetRect().left
439  arrange.sort(key=lambda x: x.GetRect().top)
440 
441  # and align all to the pivot
442  for c in arrange:
443  c.SetPosition(wx.Point(left, top))
444  top = c.GetRect().bottom + self.GetPadding()
445

769  SelectionManager is an invisible window that positions itself on the top left corner of a Board
770  and gets focus every time a card is (or many cards are) selected. This is done to hide carets
771  and selections from other controls while selection is active. SelectionManager also manages
