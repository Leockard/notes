114  def MoveBy(self, dx, dy):
115  """
116  Moves the card by the offsets dx, dy. Unlike SetPosition() and Move(),
117  this method preserves the underlying float coordinates, if any. See Card.frect.
118  """
119  bd = self.GetParent()
120  if not self.frect:
121  self.ResetFRect()
122 
123  # self.frect stores the aboslute coordinates
124  abs_left = self.frect[0] + dx
125  abs_top = self.frect[1] + dy
126 
127  # but Move() is expecting coordinates relative to the start of the view port
128  start = self.GetParent().GetViewStartPixels()
129  rel_left = abs_left - start[0]
130  rel_top = abs_top - start[1]
131 
132  self.frect = (abs_left, abs_top, self.frect[2], self.frect[3])
133  super(Card, self).Move((rel_left, rel_top))

135  def Stretch(self, factor):
136  # if we haven't stored our float coordinates yet
137  if not self.frect:
138  self.ResetFRect()
139 
140  # only scale if we're a sensitive factor away from 1.0
141  if abs(factor - 1.0) < 0.001:
142  return
143 
144  # keep count of the scale after every stretch
145  self.scale *= factor
146 
147  # wx.Rect() converts everything to integers
148  # since factor is always a float, this is why
149  # we need to store our own coordinates!
150  self.frect = [f * factor for f in self.frect]
151  self.SetRect(wx.Rect(*self.frect))
152 
153  def SetScale(self, new_scale):
154  self.Stretch(new_scale / self.scale)
