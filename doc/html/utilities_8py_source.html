47  def FitToChildren(self):
48  """
49  Call to set the virtual (content) size to tightyly fit the children.
50  If there are no children, keeps the virtual size as it is (don't shrink).
51  """
52  children = self.GetChildren()
53  if len(children) == 0: return
54 
55  # set view start at (0,0) to get absolute cordinates
56  shown = self.IsShown()
57  if shown: self.Hide()
58  view = self.GetViewStart()
59  self.Scroll(0, 0)
60 
61  # calculate children extension
62  rects = [c.GetRect() for c in self.GetChildren()]
63  right = max(rects, key=lambda r: r.right).right
64  bottom = max(rects, key=lambda r: r.bottom).bottom
65 
66  # compare and update
67  sz = self.content_sz
68  if right > sz.x: sz = wx.Size(right, sz.y)
69  if bottom > sz.y: sz = wx.Size(sz.x, bottom)
70  self.content_sz = sz
71  self.SetVirtualSize(self.content_sz)
72 
73  # return to the previous scroll position
74  self.Scroll(view[0], view[1])
75  if shown: self.Show()

102  def SetBackgroundColour(self, new_cl):
103  # If we change background colour from A to B, but a char in the text
104  # has background colour C, TextCtrl.SetBackgroundColour() won't change
105  # it correctly. Solution: store the bg colour (of those chars that have
106  # a different bg colour than the current one), change the bg for all
107  # and then restore the ones saved.
108  text = self.GetValue()
109  attr = wx.TextAttr()
110  cur = self.GetBackgroundColour()
111  char_old_bg = {}
112 
113  # store those bg's different than the current
114  for i in range(len(text)):
115  self.GetStyle(i, attr)
116  old = attr.GetBackgroundColour()
117  if old != cur:
118  char_old_bg[i] = old
119 
120  # set the new bg for all, but don't use attr again!
121  # char_old_bg is pointing to one of its members
122  super(ColouredText, self).SetBackgroundColour(new_cl)
123  self.SetStyle(0, len(text), wx.TextAttr(None, new_cl))
124 
125  # restore the saved ones
126  for i in char_old_bg.keys():
127  attr.SetBackgroundColour(char_old_bg[i])
128  self.SetStyle(i, i+1, attr)
