64  def Search(self, ev):
65  """
66  Search the current text in the search bar in all of the Cards' texts.
67  Cycle through finds with ctrl + G.
68  """
69  # search string in lower case
70  s = self.search_ctrl.GetValue().lower()
71 
72  # if we were already searching, clear up highlighting
73  if self.search_find:
74  for c, i in self.search_find:
75  c.SetStyle(i, i + len(self.search_str), c.GetDefaultStyle())
76 
77  # if no search string, reset variables and quit
78  if not s:
79  self.search_ctrl.SetBackgroundColour(wx.WHITE)
80  self.search_find = []
81  self.search_str = ""
82  self.search_head = None
83  return
84 
85  # where are we searching?
86  cards = []
87  content = self.GetCurrentPage().GetCurrentContent()
88  if content == Board:
89  cards = self.GetCurrentBoard().GetCards()
90  elif content == CardInspect:
91  cards = self.GetCurrentPage().view_card.GetCards()
92 
93  # gather all (lower case) values in which to search
94  # including the control they appear in
95  txt_ctrls = []
96  for c in cards:
97  if isinstance(c, Content):
98  txt_ctrls.append((c.GetTitle().lower(), c.title))
99  txt_ctrls.append((c.GetContent().lower(), c.content))
100  if isinstance(c, Header):
101  txt_ctrls.append((c.GetHeader().lower(), c.header))
102 
103  # do the actual searching
104  finds = []
105  for txt, ctrl in txt_ctrls:
106  pos = [m.start() for m in re.finditer(s, txt)]
107  for p in pos:
108  finds.append((ctrl, p))
109 
110  # if success: highlight and setup vars for cycling
111  if finds:
112  self.search_ctrl.SetBackgroundColour(wx.YELLOW)
113  for c, i in finds:
114  c.SetStyle(i, i + len(s), wx.TextAttr(wx.NullColour, wx.YELLOW))
115 
116  self.search_find = finds
117  self.search_str = s
118  self.search_head = 0 # when done, set to None
119 
120  # if not found: make sure variables are setup correctly too
121  else:
122  self.search_ctrl.SetBackgroundColour(wx.RED)
123  self.search_find = []
124  self.search_str = ""
125  self.search_head = None

176  def JumpSearchResults(self, old, new):
177  """
178  Unhighlights the old search result and highlights the new one.
179  old and new must be valid indices in the internal search results list.
180  This is just a convenience function for Prev- and NextSearchResult.
181  Use those ones instead.
182  """
183  s = self.search_ctrl.GetValue()
184 
185  # erase strong highlight on previous search find
186  # even if this is the first one, nothing bad will happen
187  # we'd just painting yellow again over the last one
188  ctrl = self.search_find[old][0]
189  pos = self.search_find[old][1]
190  ctrl.SetStyle(pos, pos + len(s), wx.TextAttr(None, wx.YELLOW))
191 
192  # selection and strong hightlight on current search find
193  ctrl = self.search_find[new][0]
194  pos = self.search_find[new][1]
195  ctrl.SetStyle(pos, pos + len(s), wx.TextAttr(None, wx.RED))
196 
197  # make sure the find is visible
198  card = GetCardAncestor(ctrl)
199  if card:
200  self.GetCurrentBoard().ScrollToCard(card)
201  if isinstance(card, Content):
202  if card.IsCollapsed():
203  card.Uncollapse()
204  card.ScrollToChar(pos)
